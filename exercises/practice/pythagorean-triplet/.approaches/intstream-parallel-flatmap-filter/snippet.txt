    .flatMap(a -> IntStream.rangeClosed(a, limit).parallel().mapToObj(
            b -> new PythagoreanTriplet(a, b, Math.sqrt(a * a + b * b)))
        .filter(pt -> pt.c % 1 == 0 && pt.c <= limit &&
            pt.sum() == num))
    .toList()
